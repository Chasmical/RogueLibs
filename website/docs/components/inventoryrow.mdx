# `InventoryRow`

import { useState } from 'react';
import useBaseUrl from '@docusaurus/useBaseUrl';
import BrowserWindow from '@site/src/components/BrowserWindow';

import InventorySlot from '@site/src/components/InventorySlot';
import InventoryRow from '@site/src/components/InventoryRow';

### Props

```ts
export type Props = {
  items?: SlotProps[],        // alternatively, you can add InventorySlot props here
  children?: React.ReactNode, // InventorySlot children
  width?: number,             // minimum width of the row
  interactable?: boolean,     // determines whether it's interactable or not
  onChange?: (index: number) => void, // changed selected item handler
}
```

### Typical usage

```jsx
<InventoryRow>
  <InventorySlot sprite={useBaseUrl("/img/TestItem.png")} tooltip="$70"/>
  <InventorySlot sprite={useBaseUrl("/img/TestItem.png")} count={13}/>
</InventoryRow>
```

<BrowserWindow>
<InventoryRow>
  <InventorySlot sprite={useBaseUrl("/img/TestItem.png")} tooltip="$70"/>
  <InventorySlot sprite={useBaseUrl("/img/TestItem.png")} count={13}/>
</InventoryRow>
</BrowserWindow>

### Using `items` instead

:::note
It's a tiny bit better, in terms of performance, but you should use `InventorySlot` children anyway, when possible.
:::

```jsx
<InventoryRow items={[
    {sprite: useBaseUrl("/img/TestItem.png"), tooltip: "$70"},
    {sprite: useBaseUrl("/img/TestItem.png"), count: 13},
  ]}/>
```

<BrowserWindow>
<InventoryRow items={[
    {sprite: useBaseUrl("/img/TestItem.png"), tooltip: "$70"},
    {sprite: useBaseUrl("/img/TestItem.png"), count: 13},
  ]}/>
</BrowserWindow>

### Set minimum width

```jsx
<InventoryRow width={5}>
  <InventorySlot sprite={useBaseUrl("/img/TestItem.png")} tooltip="$70"/>
  <InventorySlot sprite={useBaseUrl("/img/TestItem.png")} count={13}/>
</InventoryRow>
```

<BrowserWindow>
<InventoryRow width={5}>
  <InventorySlot sprite={useBaseUrl("/img/TestItem.png")} tooltip="$70"/>

</BrowserWindow>

### Interactable

```jsx
<InventoryRow width={5} interactable={true}>
  <InventorySlot sprite={useBaseUrl("/img/TestItem.png")} tooltip="$70"/>
  <InventorySlot sprite={useBaseUrl("/img/TestItem.png")} count={13}/>
  <InventorySlot sprite={useBaseUrl("/img/TestItem.png")} count={77} countColor="#55BB11"/>
</InventoryRow>
```
