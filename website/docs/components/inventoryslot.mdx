# `InventorySlot`

import { useState } from 'react';
import useBaseUrl from '@docusaurus/useBaseUrl';
import BrowserWindow from '@site/src/components/BrowserWindow';

import InventorySlot from '@site/src/components/InventorySlot';

### Props

```ts
export type Props = {
  sprite?: string,       // item's sprite
  tooltip?: string,      // item's tooltip text
  tooltipColor?: string, // item's tooltip color
  count?: string,        // item's count text
  countColor?: string,   // item's count color
  type?: "normal" | "selected" | "locked" | null,
    // if set to any string, the slot will react to the cursor
  onClick?: () => void,  // click handler
}
```

### Typical usage

```jsx
<InventorySlot sprite={useBaseUrl("/img/TestItem.png")}
  count={13} tooltip="$70"/>
```

<BrowserWindow>
<InventorySlot sprite={useBaseUrl("/img/TestItem.png")}
  count={13} tooltip="$70"/>
</BrowserWindow>

### With colors

```jsx
<InventorySlot sprite={useBaseUrl("/img/TestItem.png")}
  count={13} countColor="#55BB11"
  tooltip="$70" tooltipColor="#2299FF66"/>
```

<BrowserWindow>
<InventorySlot sprite={useBaseUrl("/img/TestItem.png")}
  count={13} countColor="#55BB11"
  tooltip="$70" tooltipColor="#2299FF66"/>
</BrowserWindow>

### Hoverable

Set `type` to `"normal"`, `"selected"` or `"locked"`.

:::info
It's used inside other components, like [`InventoryRow`](./inventoryrow) and [`InventoryGrid`](./inventorygrid).
:::

```jsx
<InventorySlot sprite={useBaseUrl("/img/TestItem.png")}
  count={13} tooltip="$70" type="normal"/>
```

<BrowserWindow>
<InventorySlot sprite={useBaseUrl("/img/TestItem.png")}
  count={13} tooltip="$70" type="normal"/>
</BrowserWindow>
